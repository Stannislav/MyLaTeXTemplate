\bookmarksetup{
  open,
  numbered,
  addtohook={% CHANGE THE LEVELS DEPENDING ON WHETHER CHAPTER OR SECTION IS THE LOWEST LEVEL
    \ifnum\bookmarkget{level}<1 %
      \bookmarksetup{bold}%
    \fi%
    \ifnum\bookmarkget{level}>1 %
      \bookmarksetup{italic}%
    \fi
  },
}
